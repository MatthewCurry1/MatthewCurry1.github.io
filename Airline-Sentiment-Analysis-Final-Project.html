# Use VectorSource() to make each element a vector
tweets_source <- VectorSource(tweets$text)

# Use VCorpus() to store newly created object in RAM
tweets_corpus <- VCorpus(tweets_source)

# Clean tweets_corpus for better analysis
tweets_corpus <- tm_map(tweets_corpus, removeNumbers)
tweets_corpus <- tm_map(tweets_corpus, removePunctuation)
tweets_corpus <- tm_map(tweets_corpus, content_transformer(tolower))

Analysis

# Find the amount of positive and negative words per tweet
positive_count <- rowSums(tweets_m[, colnames(tweets_m) %in% p])
negative_count <- rowSums(tweets_m[, colnames(tweets_m) %in% n])

# Difference between the amount of positive and negative words per tweet
tweets$opinion <- positive_count - negative_count
# Analyzes the tweets into the new table and measures the average opinion per airline,
# confidence in whether tweet is positive, neutral, or negative per airline,
# and percent of the three per airline 
airline_opinion <- tweets %>%
  group_by(airline) %>%
  summarise(
    average_opinion = mean(opinion, na.rm = TRUE),
    confidence = mean(airline_sentiment_confidence),
    positive_percent = mean(airline_sentiment == "positive"),
    neutral_percent = mean(airline_sentiment == "neutral"),
    negative_percent = mean(airline_sentiment == "negative")
  )

# Adds average delay and cancel percent to combined_analysis, where the airline names match
combined_analysis <- airline_opinion %>%
  inner_join(
    flightdata %>%
      group_by(AIRLINE_NAME) %>%
      summarise(
        average_delay = mean(ARRIVAL_DELAY, na.rm = TRUE),
        cancel_percent = (mean(CANCELLED, na.rm = TRUE) * 100)
      ),
    by = c("airline" = "AIRLINE_NAME")
  )

# Create table to measure average delay and cancel percent per airline
flight_summary <- flightdata %>%
  group_by(AIRLINE_NAME) %>%
  summarise(
    average_delay = mean(ARRIVAL_DELAY, na.rm = TRUE),
    cancel_percent = mean(CANCELLED, na.rm = TRUE) * 100
  )

# Create table that shows positive, neutral, and negative count and percentage by airline
sentiment_summary <- tweets %>%
  group_by(airline, airline_sentiment) %>%
  summarise(count = n(), .groups = "drop") %>%
  group_by(airline) %>%
  mutate(percentage = count / sum(count) * 100)

Conclusion

Our analysis suggests that airlines receiving more negative sentiment on Twitter also tend to have higher cancellation rates and longer delays. While sentiment cannot perfectly predict performance, the patterns indicate that public opinion often reflects real operational issues. This project demonstrates how text mining and flight data can be combined to explore perception versus performance in the airline industry, though we’re sure it is not limited to just that.
